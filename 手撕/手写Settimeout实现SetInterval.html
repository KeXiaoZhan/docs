# 手写setTimeout

特点: 参数为回调函数，时间，传入回到函数的参数

思路：

  • 使用new Date获取创建定时器时间start，维护一个定时器变量和持续变化的时间
  • 使用window.requestAnimationFrame 在每次动画执行前更新时间
  • 到达时间后取消动画帧 window.cancelAnimationFrame
	let mySetTimeout = (fn, timeout, ...args) => {
        let start = +new Date();
        let timer, now;
        const loop = () => {
            //每当浏览器开始绘制新的一帧画面的时候，都会去执行一下requestAnimationFrame,肉眼之所以能看到页面就是一帧帧动画在不断刷着
            timer = window.requestAnimationFrame(loop);
            now = +new Date();
            if (now - start >= timeout) {
                fn.apply(null, args);
                window.cancelAnimationFrame(timer);
            }
        }
        //做到异步，定时器是一个异步任务
        window.requestAnimationFrame(loop);
    }
    function say() {
        console.log("hello")
    }
    mySetTimeout(say, 1000)

# 模拟setInterval

思路：

  • 在setInterval内递归调用setTimeout，不能直接调用回调函数fn，需要在interval内部再定义一个函数来使用，不然递归不了
function mySetInterval(fn, timeout,...args) {
    function inside() {
        fn.apply(null,args);
        mySetTimeout(inside, timeout,...args)
    }
    mySetTimeout(inside, timeout,...args)
}
mySetInterval(say,1000,"a", "b", "c")
